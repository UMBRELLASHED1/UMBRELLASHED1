// ============================================================ // SMART UMBRELLA SYSTEM - ESP32 Code (UPDATED) // with Firebase Real-time Database Integration // ============================================================ #include #include #include #include "addons/TokenHelper.h" #include "addons/RTDBHelper.h" // ================= DHT ================= #define DHT_PIN 4 #define DHTTYPE DHT22 DHT dht(DHT_PIN, DHTTYPE); // ================= WIFI ================= const char* ssid = "fh_b4a558"; const char* password = "wlan4b5aa7"; // ================= FIREBASE ================= #define API_KEY "AIzaSyA9KGSzIxg05Esja156k8pLwnJlLysYHRI" #define DATABASE_URL "umbrellashednew-default-rtdb.asia-southeast1.firebasedatabase.app" // ================= MOTORS ================= #define STEP_PIN 25 #define DIR_PIN 26 #define ENA_PIN 27 #define STEP2_PIN 32 #define DIR2_PIN 33 #define ENA2_PIN 14 // Set this to LOW if your stepper driver enable pin is active low // (most A4988/DRV8825 boards use LOW to enable the motor) #define ENA_ACTIVE_LEVEL LOW // ================= SENSORS ================= #define LDR_PIN 18 #define LDR_PIN2 5 #define LDR_PIN3 12 #define LDR_PIN4 13 #define RAIN_PIN 19 // ================= ACTUATORS ================= #define LED_PIN 23 // ================= VARIABLES ================= bool isOpen = false; String lastCommand = ""; FirebaseData fbdo; FirebaseConfig config; FirebaseAuth auth; unsigned long lastSendTime = 0; const long SEND_INTERVAL = 3000; unsigned long lastReadTime = 0; const long READ_INTERVAL = 1000; unsigned long lastModeReadTime = 0; const long MODE_READ_INTERVAL = 2000; unsigned long lastAutoCheckTime = 0; const long AUTO_CHECK_INTERVAL = 2000; bool isAutoMode = true; // Start in auto mode // ================= SETUP ================= void setup() { Serial.begin(115200); delay(1000); dht.begin(); pinMode(STEP_PIN, OUTPUT); pinMode(DIR_PIN, OUTPUT); pinMode(ENA_PIN, OUTPUT); pinMode(STEP2_PIN, OUTPUT); pinMode(DIR2_PIN, OUTPUT); pinMode(ENA2_PIN, OUTPUT); pinMode(LDR_PIN, INPUT); pinMode(RAIN_PIN, INPUT); pinMode(LED_PIN, OUTPUT); digitalWrite(ENA_PIN, !ENA_ACTIVE_LEVEL); digitalWrite(ENA2_PIN, !ENA_ACTIVE_LEVEL); digitalWrite(LED_PIN, HIGH); Serial.println("\n=== ESP32 Umbrella System Starting ==="); // WIFI WiFi.begin(ssid, password); Serial.print("Connecting to WiFi"); int wifiAttempts = 0; while (WiFi.status() != WL_CONNECTED && wifiAttempts < 20) { Serial.print("."); delay(500); wifiAttempts++; } if (WiFi.status() == WL_CONNECTED) { Serial.println("\n✅ WiFi Connected!"); Serial.println(WiFi.localIP()); } else { Serial.println("\n❌ WiFi Connection Failed!"); } // FIREBASE config.database_url = DATABASE_URL; config.signer.test_mode = true; Firebase.reconnectNetwork(true); fbdo.setBSSLBufferSize(4096, 1024); fbdo.setResponseSize(2048); Firebase.begin(&config, &auth); Serial.println("🔥 Initializing Firebase..."); delay(2000); // Wait for Firebase to be ready int fbAttempts = 0; while (!Firebase.ready() && fbAttempts < 20) { Serial.print("."); delay(500); fbAttempts++; } if (Firebase.ready()) { Serial.println("\n✅ Firebase Connected!"); if (!Firebase.RTDB.setString(&fbdo, "/umbrellaState", "CLOSED")) { Serial.println("❌ Init setString failed: " + fbdo.errorReason()); } if (!Firebase.RTDB.setString(&fbdo, "/control", "NONE")) { Serial.println("❌ Init setString failed: " + fbdo.errorReason()); } if (!Firebase.RTDB.setString(&fbdo, "/mode", "auto")) { Serial.println("❌ Init setString failed: " + fbdo.errorReason()); } } else { Serial.println("\n❌ Firebase initialization failed!"); Serial.println("Error reason: " + fbdo.errorReason()); } } // ================= SEND SENSOR DATA ================= void sendSensorData() { if (millis() - lastSendTime < SEND_INTERVAL) return; lastSendTime = millis(); int lightRaw = digitalRead(LDR_PIN); int rainRaw = digitalRead(RAIN_PIN); float temp = dht.readTemperature(); float hum = dht.readHumidity(); String lightStatus = (lightRaw == LOW) ? "LIGHT PRESENT" : "NO LIGHT PRESENT"; String rainStatus = (rainRaw == LOW) ? "RAIN DETECTED" : "NO RAIN DETECTED"; Serial.println("\n📊 SENSOR DATA"); Serial.print("🌡 TEMP: "); Serial.println(temp); Serial.print("💧 HUM: "); Serial.println(hum); Serial.print("💡 LIGHT: "); Serial.println(lightStatus); Serial.print("🌧 RAIN: "); Serial.println(rainStatus); // 🔥 SEND TO FIREBASE if (Firebase.ready()) { Serial.println("🔥 Sending to Firebase..."); if (!isnan(temp)) { if (!Firebase.RTDB.setFloat(&fbdo, "/temperature", temp)) { Serial.println("❌ Temp Error: " + fbdo.errorReason()); } } if (!isnan(hum)) { if (!Firebase.RTDB.setFloat(&fbdo, "/humidity", hum)) { Serial.println("❌ Hum Error: " + fbdo.errorReason()); } } if (!Firebase.RTDB.setString(&fbdo, "/lightStatus", lightStatus)) { Serial.println("❌ Light Error: " + fbdo.errorReason()); } if (!Firebase.RTDB.setString(&fbdo, "/rainStatus", rainStatus)) { Serial.println("❌ Rain Error: " + fbdo.errorReason()); } Serial.println("✅ Send Attempt Done"); } else { Serial.println("❌ Firebase not ready"); } } // ================= READ COMMAND (Manual Control) ================= void readCommand() { if (millis() - lastReadTime < READ_INTERVAL) return; lastReadTime = millis(); if (!Firebase.ready()) return; if (Firebase.RTDB.getString(&fbdo, "/control")) { String command = fbdo.stringData(); if (command != "NONE") { Serial.print("📩 Command: "); Serial.println(command); if (command == "OPEN") { openUmbrella(); } else if (command == "CLOSE") { closeUmbrella(); } else if (command == "LED_ON") { digitalWrite(LED_PIN, LOW); Serial.println("💡 LED ON"); } else if (command == "LED_OFF") { digitalWrite(LED_PIN, HIGH); Serial.println("💡 LED OFF"); } // Reset command after execution so the same command can be sent again later if (!Firebase.RTDB.setString(&fbdo, "/control", "NONE")) { Serial.println("❌ Failed to clear command: " + fbdo.errorReason()); } lastCommand = "NONE"; } } } void readMode() { if (millis() - lastModeReadTime < MODE_READ_INTERVAL) return; lastModeReadTime = millis(); if (!Firebase.ready()) return; if (Firebase.RTDB.getString(&fbdo, "/mode")) { String mode = fbdo.stringData(); mode.toLowerCase(); bool newAutoMode = (mode == "auto"); if (newAutoMode != isAutoMode) { isAutoMode = newAutoMode; Serial.print("⚙️ Mode changed to: "); Serial.println(mode); } } } // ================= AUTO MODE LOGIC ================= void autoModeLogic() { if (millis() - lastAutoCheckTime < AUTO_CHECK_INTERVAL) return; lastAutoCheckTime = millis(); int lightRaw = digitalRead(LDR_PIN); int rainRaw = digitalRead(RAIN_PIN); bool isLightPresent = (lightRaw == LOW); bool isRaining = (rainRaw == LOW); if (isRaining || isLightPresent) { if (!isOpen) { Serial.println("🤖 AUTO OPEN (Rain or Light Detected)"); openUmbrella(); } } else { if (isOpen) { Serial.println("🤖 AUTO CLOSE (No Rain & No Light)"); closeUmbrella(); } } } // ================= LOOP ================= void loop() { sendSensorData(); readMode(); if (isAutoMode) { autoModeLogic(); } else { readCommand(); } delay(500); } // ================= MOTOR FUNCTIONS ================= void openUmbrella() { if (isOpen) { Serial.println("⚠️ Already Open!"); return; } Serial.println("🔓 Opening Umbrella..."); digitalWrite(DIR_PIN, HIGH); digitalWrite(DIR2_PIN, HIGH); digitalWrite(ENA_PIN, ENA_ACTIVE_LEVEL); digitalWrite(ENA2_PIN, ENA_ACTIVE_LEVEL); runMotor(20000); digitalWrite(ENA_PIN, !ENA_ACTIVE_LEVEL); digitalWrite(ENA2_PIN, !ENA_ACTIVE_LEVEL); isOpen = true; Serial.println("✅ Umbrella Opened"); // Update Firebase if (Firebase.ready()) { Firebase.RTDB.setString(&fbdo, "/umbrellaState", "OPEN"); } } void closeUmbrella() { if (!isOpen) { Serial.println("⚠️ Already Closed!"); return; } Serial.println("🔒 Closing Umbrella..."); digitalWrite(DIR_PIN, LOW); digitalWrite(DIR2_PIN, LOW); digitalWrite(ENA_PIN, ENA_ACTIVE_LEVEL); digitalWrite(ENA2_PIN, ENA_ACTIVE_LEVEL); runMotor(20000); digitalWrite(ENA_PIN, !ENA_ACTIVE_LEVEL); digitalWrite(ENA2_PIN, !ENA_ACTIVE_LEVEL); isOpen = false; Serial.println("✅ Umbrella Closed"); // Update Firebase if (Firebase.ready()) { Firebase.RTDB.setString(&fbdo, "/umbrellaState", "CLOSED"); } } void runMotor(unsigned long duration) { unsigned long start = millis(); while (millis() - start < duration) { digitalWrite(STEP_PIN, HIGH); digitalWrite(STEP2_PIN, HIGH); delayMicroseconds(1500); digitalWrite(STEP_PIN, LOW); digitalWrite(STEP2_PIN, LOW); delayMicroseconds(1500); } } // ================= FIREBASE DEBUG ================= void debugFirebaseStatus() { if (Firebase.ready()) { Serial.println("✅ Firebase Connected"); } else { Serial.println("❌ Firebase Disconnected"); } }
